  <script>

    // TENDENCIAS

    fetch('https://api.rss2json.com/v1/api.json?rss_url=https://trends.google.com/trends/trendingsearches/daily/rss?geo=PY')

      .then(res => res.json())

      .then(data => {

        const trendsList = document.getElementById('trends-list');

        data.items.slice(0, 5).forEach(item => {

          let li = document.createElement('li');

          li.innerHTML = `<a href="${item.link}" target="_blank">${item.title}</a>`;

          trendsList.appendChild(li);

        });

      });

    // CLIMA

    const apiKey = 'TU_API_KEY_OPENWEATHERMAP'; // registrate en openweathermap.org y pon tu key

    fetch(`https://api.openweathermap.org/data/2.5/weather?q=Asuncion,py&units=metric&appid=${apiKey}`)

      .then(res => res.json())

      .then(data => {

        document.getElementById('weather-info').textContent =

          `Temp: ${data.main.temp}°C, Estado: ${data.weather[0].description}`;

      });
